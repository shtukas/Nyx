NxPoints are nodes of the Nyx Network. They are marbles on disk.

NxPoint (Common)
    - uuid            : String
    - unixtime        : Integer
    - nxType          : "NxNav" | "Line" | "Url" | "Text" | "UniqueFileClickable" | "FSLocation" | "FSUniqueString"
    - description     : String
    - links           : Array[SpaceId] 
                            # setid = "links:b645d07ddd5"
                            # Marbles::addSetData(filepath, "links:b645d07ddd5", link, link)

NxNav:
    - nxType : "NxNav"

NxLine:
    - nxType : "Line"

NxUrl:
    - nxType : "Url"
    - url    : String

NxText:
    - nxType : "Text"
    - text   : String # The actual text

NxUniqueFileClickable:
    - nxType : "UniqueFileClickable"
    - nhash  : String # AionPoint (one file)

NxFSLocation:
    - nxType : "FSLocation"

NxFSUniqueString:
    - nxType       : "FSUniqueString"
    - uniquestring : String
